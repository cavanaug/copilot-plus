<!-- General-purpose -- key convention (NO hardcoded mapping table) -->
Any top-level key in config.json that starts with "--" is treated as a CLI flag:
  "--allow-tool": ["shell(git:*)", "write"]  →  --allow-tool 'shell(git:*)' --allow-tool write
  "--yolo": true                              →  --yolo
  "--yolo": false                             →  (skipped)
  "--model": "gpt-4.1"                        →  --model gpt-4.1
  "--max-autopilot-continues": 5              →  --max-autopilot-continues 5
  "model": "gpt-4.1"   (no -- prefix)        →  (silently ignored)
  "--bad": {}          (object)               →  exit 1 with error to stderr
  "--bad": null        (null)                 →  exit 1 with error to stderr

<!-- Value type dispatch -->
type == "array"   → one --key value pair per element (strings/numbers only; error on non-scalar elements)
type == "boolean" and value == true  → bare --key flag (no value)
type == "boolean" and value == false → skip entirely
type == "string"  → --key value (single pair)
type == "number"  → --key value (single pair, value as string)
type == "object"  → exit 1 with error to stderr
type == "null"    → exit 1 with error to stderr

<!-- bats test structure reference -->

#   teardown() — runs after each test
#   @test "description" { run <command>; [ "$status" -eq 0 ]; }
#   BATS_TMPDIR — writable temp dir per test run

<task type="auto" tdd="true">
  <name>Task 2: GREEN — Implement the copilot wrapper script</name>
  <files>copilot</files>
  <action>
    Create `copilot` (no extension) in the repo root. Must be chmod +x.

    **Script skeleton:**
    ```bash
    #!/usr/bin/env bash
    set -euo pipefail

    REAL_COPILOT=/home/linuxbrew/.linuxbrew/bin/copilot
    CONFIG_FILE="${HOME}/.copilot/config.json"
    config_flags=()

    if [[ -f "$CONFIG_FILE" ]] && [[ -r "$CONFIG_FILE" ]]; then
      # Validate JSON first
      if ! jq empty "$CONFIG_FILE" 2>/dev/null; then
        echo "copilot-wrapper: error: invalid JSON in $CONFIG_FILE" >&2
        exit 1
      fi

      # Iterate over all top-level keys that start with "--"
      while IFS= read -r key; do
        # Get the JSON type of this key's value
        vtype=$(jq -r --arg k "$key" '.[$k] | type' "$CONFIG_FILE")

        case "$vtype" in
          array)
            while IFS= read -r element; do
              config_flags+=("$key" "$element")
            done < <(jq -r --arg k "$key" '.[$k][]' "$CONFIG_FILE")
            ;;
          boolean)
            val=$(jq -r --arg k "$key" '.[$k]' "$CONFIG_FILE")
            if [[ "$val" == "true" ]]; then
              config_flags+=("$key")
            fi
            # false → skip entirely
            ;;
          string|number)
            val=$(jq -r --arg k "$key" '.[$k]' "$CONFIG_FILE")
            config_flags+=("$key" "$val")
            ;;
          object|null|*)
            echo "copilot-wrapper: error: unsupported value type '$vtype' for key '$key' in $CONFIG_FILE" >&2
            exit 1
            ;;
        esac
      done < <(jq -r 'keys[] | select(startswith("--"))' "$CONFIG_FILE")
    fi
    # If config missing or unreadable: config_flags stays empty, proceed silently

    exec "$REAL_COPILOT" "${config_flags[@]+"${config_flags[@]}"}" "$@"
    ```

    Notes:
    - `jq -r 'keys[] | select(startswith("--"))'` extracts only `--`-prefixed keys
    - `"${config_flags[@]+"${config_flags[@]}"}` handles empty array with `set -u`
    - `exec` replaces the wrapper process — no subprocess overhead
    - Error messages prefixed with `copilot-wrapper: error:` for clarity
    - Non-`--` keys are never iterated (the `select(startswith("--"))` filter handles MAP-07)

    After writing, run `chmod +x copilot`.
    Then run `bats tests/copilot.bats` — ALL tests must PASS (GREEN phase).
